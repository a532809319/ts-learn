        this._Sex = Sex;

        return "大家好，我是" + person.Name;
    }
  private  _Name: string;
  private  _Sex: boolean;
  Age: string;
  get Sex() {
      return this._Sex;

      this._Sex = Sex;
  }
  get Name() {
      return this._Name;
  }
  set Name(Name:string) {
      this._Name = Name;
  }

}
